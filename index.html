<script>

/* AUTO DJ + AUTO COVER */
let tracks=[];
let covers={};
let i=0;

fetch("https://api.github.com/repos/noirsec/Winhart/contents/")
.then(r=>r.json())
.then(data=>{
data.forEach(f=>{
let name=f.name.replace(/\.(mp3|jpg)/i,"");

if(f.name.toLowerCase().endsWith(".mp3")){
tracks.push({title:name,url:f.download_url});
}

if(f.name.toLowerCase().endsWith(".jpg")){
covers[name]=f.download_url;
}
});
title.innerText="Klik Play";
});

const player=document.getElementById("player");

/* PLAY */
function playRadio(){
if(!tracks.length)return;

let track=tracks[i];
player.src=track.url;
player.play().catch(()=>{});

title.innerText=track.title;

/* SET COVER */
if(covers[track.title]){
cover.src=covers[track.title];
}else{
cover.src="";
}
}

function pauseRadio(){player.pause();}

/* NEXT AUTO */
player.addEventListener("ended",()=>{
i++;
if(i>=tracks.length)i=0;
playRadio();
});

/* PROGRESS */
player.ontimeupdate=()=>{
let p=(player.currentTime/player.duration)*100;
progress.style.width=p+"%";
};
